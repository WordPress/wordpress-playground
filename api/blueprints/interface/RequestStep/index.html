RequestStep

Sends a HTTP request to the Playground.

@needsLogin
@hasRunnableExample
@example
<code>
{
        "step": "request",
        "request": {
            "method": "POST",
            "url": "/wp-admin/admin-ajax.php",
            "formData": {
                "action": "my_action",
                "foo": "bar"
            }
        }
}
</code>
